дәәи„ёиҜҶеҲ«з”ЁеҲ°зҡ„ж•°жҚ®жЁЎеһӢ

еҹәдәҺзӣ®еүҚзҡ„ж•°жҚ®йҮҸе’ҢдёӘдәәзҹҘиҜҶдёҺиғҪеҠӣпјҢжҲ‘иҝҳдёҚиғҪиҮӘе·ұеҲ©з”ЁжңәеҷЁеӯҰд№ жқҘи®ӯз»ғж•°жҚ®жЁЎеһӢпјҢжүҖд»ҘдҪҝз”Ёзҡ„жҳҜеӣҪеӨ–еҲ«дәәе·Із»Ҹи®ӯз»ғеҘҪзҡ„ж•°жҚ®жЁЎеһӢгҖӮ

1. OpenCVдҪҝз”ЁеҲ°зҡ„дәәи„ёеҲҶзұ»жЁЎеһӢxmlпјҡhaarcascades

дёҠйқўй“ҫжҺҘдёӯжүҖжңүзҡ„xmlйғҪжҺЁиҚҗдёӢиҪҪеҲ°жң¬ең°еӨҮз”ЁпјҢжҲ‘жүҖдҪҝз”Ёзҡ„жҳҜhaarcascade_frontalface_default.xml

2. DlibдҪҝз”ЁеҲ°зҡ„дәәи„ёиҜҶеҲ«и®ӯз»ғжЁЎеһӢпјҡshape_predictor

жҲ‘жүҖдҪҝз”Ёзҡ„жҳҜshape_predictor_68_face_landmarks.dat.bz2е’Ңshape_predictor_5_face_landmarks.dat.bz2иҝҷдёӨдёӘгҖӮ

жҺЁиҚҗеӨ§е®¶жҠҠиҝҷдёӨдёӘйғҪдёӢиҪҪдёӢжқҘпјҢе№¶и§ЈеҺӢпјҢжҲ‘д»¬жүҖйңҖиҰҒзҡ„жҳҜйҮҢйқўзҡ„datж•°жҚ®ж–Үд»¶гҖӮ

еҘҪдәҶпјҢжңүдәҶдёҠйқўзҡ„иҝҷдәӣеҹәзЎҖзҺҜеўғдёҺдҫқиө–пјҢжҲ‘д»¬еҸҜд»Ҙз»§з»ӯжҲ‘д»¬жҺҘдёӢжқҘзҡ„з®ҖеҚ•дәәи„ёжЈҖжөӢдёҺиҜҶеҲ«дәҶгҖӮ жІЎжңүиЈ…еҘҪиҝҷдәӣзҺҜеўғзҡ„еҗҢеӯҰиҜ·жіЁж„ҸдәҶпјҢжҲ‘д»¬зҸӯзә§е·Із»ҸдёҺи°·жӯҢгҖҒзҷҫеәҰзӯүеӨ§е…¬еҸёиҫҫжҲҗеҗҲдҪңпјҢжңүд»»дҪ•й—®йўҳиҜ·еңЁGoogleе’ҢBaiduиҝӣиЎҢжҗңзҙўпјҢд»–д»¬дјҡжҸҗдҫӣдҪ жүҖйңҖиҰҒзҡ„зӯ”жЎҲгҖӮ

дәҢгҖҒз®ҖеҚ•дәәи„ёжЈҖжөӢ

еңЁиҝҷдёҖиҠӮдёҚдјҡи®Іи§ЈеӨҚжқӮзҡ„дәәи„ёиҜҶеҲ«пјҢеҸӘжҳҜе…ҲйҖҡиҝҮз®ҖеҚ•зҡ„е°ҸдҫӢеӯҗи®©еӨ§е®¶е…ҲдәҶи§ЈдёҖдёӢдәәи„ёиҜҶеҲ«йңҖиҰҒз”ЁеҲ°д»Җд№ҲгҖҒдәәи„ёиҜҶеҲ«зҡ„еҹәжң¬жӯҘйӘӨзӯүзӯүжңҖз®ҖеҚ•зҡ„еҶ…е®№гҖӮ

1. еҲ©з”ЁOpenCVиҝӣиЎҢеӣҫзүҮдәәи„ёжЈҖжөӢе№¶иҝӣиЎҢж ҮжіЁ

import cv2

filepath = "img/sdu/6.jpg"

classifier = cv2.CascadeClassifier(
    "lib/opencv/haarcascades/haarcascade_frontalface_default.xml"
)
img = cv2.imread(filepath)  # иҜ»еҸ–еӣҫзүҮ
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)  # зҒ°еәҰеҢ–
color = (0, 255, 0)  # е®ҡд№үз»ҳеҲ¶йўңиүІ

# и°ғз”ЁиҜҶеҲ«дәәи„ё
faceRects = classifier.detectMultiScale(
    gray, scaleFactor=1.2, minNeighbors=3, minSize=(32, 32))
if len(faceRects):  # еӨ§дәҺ0еҲҷжЈҖжөӢеҲ°дәәи„ё
    for faceRect in faceRects:  # еҚ•зӢ¬жЎҶеҮәжҜҸдёҖеј дәәи„ё
        x, y, w, h = faceRect
        # жЎҶеҮәдәәи„ё
        cv2.rectangle(img, (x, y), (x + h, y + w), color, 2)

cv2.imshow("image", img)
cv2.waitKey(0)
cv2.destroyAllWindows()

import cv2

def scan(img):
    grayImg = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    classifier = cv2.CascadeClassifier(
        "lib/opencv/haarcascades/haarcascade_frontalface_default.xml"
    )
    color = (0, 255, 0)  # е®ҡд№үз»ҳеҲ¶йўңиүІ
    faceRects = classifier.detectMultiScale(
        grayImg, scaleFactor=1.2, minNeighbors=3, minSize=(32, 32))
    if len(faceRects):  # еӨ§дәҺ0еҲҷжЈҖжөӢеҲ°дәәи„ё
        for faceRect in faceRects:  # еҚ•зӢ¬жЎҶеҮәжҜҸдёҖеј дәәи„ё
            x, y, w, h = faceRect
            cv2.rectangle(img, (x, y), (x + h, y + w), color, 2)
    cv2.imshow("image", img)

cap = cv2.VideoCapture(0)
while (1):
    ret, frame = cap.read()
    scan(frame)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()

import cv2
import dlib

path = "img/sdu/658.jpg"
img = cv2.imread(path)
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

#дәәи„ёеҲҶзұ»еҷЁ
detector = dlib.get_frontal_face_detector()

dets = detector(gray, 1)
for face in dets:
    #еңЁеӣҫзүҮдёӯж ҮжіЁдәәи„ёиҫ№жЎҶпјҢе№¶жҳҫзӨә
    left = face.left()
    top = face.top()
    right = face.right()
    bottom = face.bottom()
    cv2.rectangle(img, (left, top), (right, bottom), (0, 255, 0), 2)
    cv2.imshow("image", img)

cv2.waitKey(0)
cv2.destroyAllWindows()

еӣӣгҖҒеҲ©з”ЁGPUеҠ йҖҹDlibжҸҗй«ҳж•ҲзҺҮ

еңЁеүҚйқўзҡ„е°Ҹе®һйӘҢдёӯжҲ‘д»¬еҸ‘зҺ°дәҶпјҢдҪҝз”ЁDlibжқҘе®һзҺ°зҡ„дәәи„ёжЈҖжөӢдёҺиҜҶеҲ«йҖҹеәҰеҫҲж…ўпјҢеңЁи§Ҷйў‘дёӯеҫҲйҡҫе®һзҺ°е®һж—¶жҖ§гҖӮ жүҖд»ҘжҲ‘д»¬дёҚд»…иҰҒжғіпјҢжҳҜеҗҰеҸҜд»Ҙз”ЁGPUеҠ йҖҹдёҖдёӢгҖӮ д»ҘеүҚеҸҜиғҪеҫҲйә»зғҰпјҢзҺ°еңЁиӢұдјҹиҫҫжҺЁеҮәзҡ„CUDAе°ұдё“й—ЁдёәдәҶи§ЈеҶіиҝҷдёӘиҖҢз”ҹгҖӮ

1. йҰ–е…ҲдёӢиҪҪCUDA

CUDAдёӢиҪҪй“ҫжҺҘ

дёӢиҪҪе№¶е®үиЈ…пјҢдёҚеҝ…еӨҡиҜҙпјҢдёҖи·ҜдёӢдёҖжӯҘ

2. з„¶еҗҺдёӢиҪҪcudnn

cudnnдёӢиҪҪй“ҫжҺҘ

иҝҷйҮҢдёӢиҪҪйңҖиҰҒжіЁеҶҢиӢұдјҹиҫҫиҙҰеҸ·пјҢжіЁеҶҢе°ұиЎҢпјҢдёӢиҪҪдёӢжқҘеҗҺи§ЈеҺӢпјҢйҮҢйқўжҸҗдҫӣcudnnеә“ж–Үд»¶д»ҘеҸҠеӨҙж–Үд»¶пјҢйңҖиҰҒжҠҠcudnnзӣ®еҪ•дёӢзҡ„binгҖҒincludeд»ҘеҸҠlibзӣ®еҪ•дёӯзҡ„ж–Үд»¶жӢ·иҙқеҲ°зӣёеә”зҡ„cudaзӣ®еҪ•дёӯзҡ„binгҖҒincludeе’Ңlibзӣ®еҪ•дёӯеҺ»гҖӮ

3. е®үиЈ…cmake

cmakeжҳҜиҰҒеңЁеҗҺйқўзј–иҜ‘Dlibз”ЁеҲ°зҡ„гҖӮ cmakeдёӢиҪҪй“ҫжҺҘ и®°дҪҸе®үиЈ…е№¶ж·»еҠ pathгҖӮ

4. зј–иҜ‘Dlib

иҝҷйҮҢжҲ‘д»¬и·іиҝҮпјҢеӣ дёәеҗҺйқўеҸҜд»ҘзӣҙжҺҘеңЁе®үиЈ…Dlibеә“ж—¶и®©д»–иҮӘеҠЁзј–иҜ‘пјҢиғҪиҮӘеҠЁжҗһдҪ•еҝ…еҺ»йә»зғҰпјҢеҰӮжһңж„ҹе…ҙи¶ЈеҸҜд»ҘиҮӘе·ұеҺ»зј–иҜ‘гҖӮ

5. е®үиЈ…ж”ҜжҢҒCUDAзҡ„Dlibеә“пјҲдјҡиҮӘеҠЁзј–иҜ‘пјү

жҲ‘д»¬жү“ејҖ64дҪҚзҡ„cmdпјҲиҝҷдёҖдёӘдјҡеңЁе®үиЈ…VS2017зҡ„ж—¶еҖҷжҸҗдҫӣпјүпјҢжҲ–иҖ…жү“ејҖ64дҪҚзҡ„PowershellпјҢеңЁйҮҢйқўиҝҗиЎҢе®үиЈ…е‘Ҫд»Ө

python setup.py install --yes DLIB_USE_CUDA
